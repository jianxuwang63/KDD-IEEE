\definecolor{c1Top}{RGB}{48,115,170}
\definecolor{c1Bg}{RGB}{232,240,248}
\definecolor{c2Top}{RGB}{183,61,54}
\definecolor{c2Bg}{RGB}{249,231,228}
\definecolor{c3Top}{RGB}{77,139,72}
\definecolor{c3Bg}{RGB}{232,244,229}
\definecolor{c4Top}{RGB}{123,66,142}

% ============================================================
% Row 1: Standard Training
% ============================================================

% --- Case 1 ---
\node[box={c1Top}{c1Bg}] (c1) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 1: Risk Control}

    \vspace{4pt}
    $\widehat{R}(\tau)=\dfrac{\sum_{i\in S_\tau}\mathbb{I}(\hat{y}_i\neq y_i)}{|S_\tau|}$

    \vspace{2pt}
    $\tau_1^*=\inf\{\tau\in[0,1]\mid\widehat{R}(\tau)\leq\alpha\}$

    \vspace{4pt}
    $(f,g)_{\tau_1^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_1^*\\
    \text{abstain},          & \text{otherwise}
    \end{cases}$

    \vspace{6pt}
    {\footnotesize\textbf{Training:} Standard Cross-Entropy}\\[1pt]
    {\footnotesize\textbf{Calibration:} $\tau$ selected to bound risk $\leq\alpha$}
    \end{minipage}
};

% --- Case 2 ---
\node[box={c2Top}{c2Bg}, right=1.2cm of c1] (c2) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 2: FNR Control}

    \vspace{4pt}
    $\widehat{\mathrm{FNR}}(\tau)=\dfrac{\sum_{i\in S_\tau}\mathbb{I}(\hat{y}_i=0\wedge y_i=1)}{|\{i\in\mathcal{D}_{\mathrm{cal}}\mid y_i=1\}|}$

    \vspace{2pt}
    $\tau_2^*=\inf\{\tau\in[0,1]\mid\widehat{\mathrm{FNR}}(\tau)\leq\alpha\}$

    \vspace{4pt}
    $(f,g)_{\tau_2^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_2^*\\
    \text{abstain},          & \text{otherwise}
    \end{cases}$

    \vspace{6pt}
    {\footnotesize\textbf{Training:} Standard Cross-Entropy}\\[1pt]
    {\footnotesize\textbf{Calibration:} $\tau$ selected to bound FNR $\leq\alpha$}
    \end{minipage}
};

% --- Row 1 arrows ---
\draw[bigarrow={c1Top}] (c1.east) -- node[above, font=\footnotesize\bfseries, text=c1Top] {$\tau_1^*$ vs.\ $\tau_2^*$} (c2.west);

% ============================================================
% Row 2: + Cost-Sensitive Learning
% ============================================================

% --- Case 3 ---
\node[box={c3Top}{c3Bg}, below=1.8cm of c1] (c3) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 3: Risk \& FNR Control}

    \vspace{4pt}
    {\footnotesize Same risk-based threshold $\tau_1^*$ as Case~1}

    \vspace{2pt}
    {\footnotesize\textbf{+ Cost-Sensitive Learning during training}}

    \vspace{4pt}
    $\mathcal{L}_{\mathrm{cs}} = -\frac{1}{n}\sum\bigl[w_1 y_i\log\hat{p}_i + w_0(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{4pt}
    {\footnotesize $w_1>1$ penalizes false negatives,}\\[1pt]
    {\footnotesize shifting the decision boundary before calibration.}

    \vspace{4pt}
    {\footnotesize\textbf{Goal:} Satisfy risk bound while suppressing FNR.}
    \end{minipage}
};

% --- Case 4 ---
\node[box={c4Top}{c4Bg}, right=1.2cm of c3] (c4) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 4: FNR \& Coverage Control}

    \vspace{4pt}
    {\footnotesize Same FNR-based threshold $\tau_2^*$ as Case~2}

    \vspace{2pt}
    {\footnotesize\textbf{+ Cost-Sensitive Learning during training}}

    \vspace{4pt}
    $\mathcal{L}_{\mathrm{cs}} = -\frac{1}{n}\sum\bigl[w_1 y_i\log\hat{p}_i + w_0(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{4pt}
    {\footnotesize CSL gives higher confidence on positives,}\\[1pt]
    {\footnotesize allowing a less aggressive $\tau_2^*$.}

    \vspace{4pt}
    {\footnotesize\textbf{Goal:} Recover coverage lost in Case~2.}
    \end{minipage}
};
