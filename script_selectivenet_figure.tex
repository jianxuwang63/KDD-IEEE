\begin{tikzpicture}[
    font=\small,
    cbox/.style n args={2}{
        rectangle, rounded corners=6pt, draw=#1!70!black, fill=#2,

    harrow/.style={
        -{Latex[length=3mm,width=3mm]}, line width=1.2pt, #1

    },

\definecolor{c1}{RGB}{48,115,170}
\definecolor{c1bg}{RGB}{235,243,250}
\definecolor{c2}{RGB}{183,61,54}
\definecolor{c2bg}{RGB}{252,235,233}
\definecolor{c3}{RGB}{77,139,72}
\definecolor{c3bg}{RGB}{236,246,233}
\definecolor{c4}{RGB}{123,66,142}
\definecolor{c4bg}{RGB}{245,236,249}

% ============================================================
% Row 1
% ============================================================

\node[cbox={c1}{c1bg}] (c1) {
    \begin{minipage}[c][5.5cm][c]{7.2cm}\centering
    \textbf{Threshold Selection}

    \vspace{4pt}
    $\displaystyle\widehat{R}(\tau)=\frac{\sum_{i\in S_\tau}\mathbb{I}(\hat{y}_i\neq y_i)}{|S_\tau|}$
    \qquad
    $\tau_1^*=\inf\{\tau\in[0,1]\mid\widehat{R}(\tau)\leq\alpha\}$

    \vspace{8pt}
    \textbf{Decision Rule}

    \vspace{2pt}
    $(f,g)_{\tau_1^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_1^*\\[2pt]
    \text{abstain},       & \text{otherwise}
    \end{cases}$

    \vspace{10pt}
    {\footnotesize\color{c1!80!black}
    \textbf{Training:} Standard Cross-Entropy\\
    \textbf{Calibration:} $\tau$ chosen so that risk $\leq\alpha$\\
    \textbf{Limitation:} FNR can remain high (up to $20\%$) at practical $\alpha$}
    \end{minipage}
};
\node[tag={c1}] at (c1.north) {Case 1: Risk Control};

\node[cbox={c2}{c2bg}, right=2.0cm of c1] (c2) {
    \begin{minipage}[c][5.5cm][c]{7.2cm}\centering
    \textbf{Threshold Selection}

    \vspace{4pt}
    $\displaystyle\widehat{\mathrm{FNR}}(\tau)=\frac{\sum_{S_\tau}\mathbb{I}(\hat{y}_i=0\wedge y_i=1)}{|\{i\in\mathcal{D}_{\mathrm{cal}}\mid y_i=1\}|}$
    \qquad
    $\tau_2^*=\inf\{\tau\in[0,1]\mid\widehat{\mathrm{FNR}}(\tau)\leq\alpha\}$

    \vspace{8pt}
    \textbf{Decision Rule}

    \vspace{2pt}
    $(f,g)_{\tau_2^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_2^*\\[2pt]
    \text{abstain},       & \text{otherwise}
    \end{cases}$

    \vspace{10pt}
    {\footnotesize\color{c2!80!black}
    \textbf{Training:} Standard Cross-Entropy\\
    \textbf{Calibration:} $\tau$ chosen so that FNR $\leq\alpha$\\
    \textbf{Limitation:} Low coverage; $\sim$40\%--80\% at strict $\alpha$}
    \end{minipage}
};
\node[tag={c2}] at (c2.north) {Case 2: FNR Control};

\draw[harrow={c1}]
    (c1.east) -- (c2.west)
    node[midway, above=6pt, font=\footnotesize\bfseries, text=c1!85!black]
    {Risk-based $\tau_1^*$ \hspace{10pt}vs.\hspace{10pt} FNR-based $\tau_2^*$};

% ============================================================
% Row 2
% ============================================================

\node[cbox={c3}{c3bg}, below=2.0cm of c1] (c3) {
    \begin{minipage}[c][5.5cm][c]{7.2cm}\centering
    \textbf{Same risk-based threshold $\tau_1^*$ as Case~1}

    \vspace{2pt}
    \textbf{$+$ Cost-Sensitive Learning}

    \vspace{6pt}
    $\displaystyle\mathcal{L}_{\mathrm{cs}}=-\frac{1}{n}\sum_{i=1}^{n}
    \bigl[w_1\,y_i\log\hat{p}_i+w_0\,(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{4pt}
    {\footnotesize $w_1>1$ amplifies the penalty on false negatives,}\\[1pt]
    {\footnotesize shifting the decision boundary before calibration.}

    \vspace{12pt}
    {\footnotesize\color{c3!80!black}
    \textbf{Training:} CSL with class weights $[1,w_1]$\\
    \textbf{Calibration:} Same risk-based $\tau_1^*$ as Case~1\\
    \textbf{Effect:} FNR drops from $12.9\%$ to $7.2\%$ ($w_1=5$) at $\alpha=5\%$}
    \end{minipage}
};
\node[tag={c3}] at (c3.north) {Case 3: Risk \& FNR Control (Cost-Sensitive Learning)};

\node[cbox={c4}{c4bg}, right=2.0cm of c3] (c4) {
    \begin{minipage}[c][5.5cm][c]{7.2cm}\centering
    \textbf{Same FNR-based threshold $\tau_2^*$ as Case~2}

    \vspace{2pt}
    \textbf{$+$ Cost-Sensitive Learning}

    \vspace{6pt}
    $\displaystyle\mathcal{L}_{\mathrm{cs}}=-\frac{1}{n}\sum_{i=1}^{n}
    \bigl[w_1\,y_i\log\hat{p}_i+w_0\,(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{4pt}
    {\footnotesize CSL raises confidence scores on positives,}\\[1pt]
    {\footnotesize allowing a less aggressive $\tau_2^*$ at same FNR.}

    \vspace{12pt}
    {\footnotesize\color{c4!80!black}
    \textbf{Training:} CSL with class weights $[1,w_1]$\\
    \textbf{Calibration:} Same FNR-based $\tau_2^*$ as Case~2\\
    \textbf{Effect:} Coverage recovers from $79\%$ to $91\%$ ($w_1=10$) at $\alpha=5\%$}
    \end{minipage}
};
\node[tag={c4}] at (c4.north) {Case 4: FNR \& Coverage Control (Cost-Sensitive Learning)};

\draw[harrow={c3}]
    (c3.east) -- (c4.west)
    node[midway, above=6pt, font=\footnotesize\bfseries, text=c3!85!black]
    {Risk-based $\tau_1^*$ \hspace{10pt}vs.\hspace{10pt} FNR-based $\tau_2^*$};

% Vertical arrows

\draw[varrow={c4}]
    (c2.south) -- node[right=4pt, font=\footnotesize\bfseries, text=c4!80!black, pos=0.5]
    {\shortstack{+ Cost-Sensitive\\\ Learning}} (c4.north);
