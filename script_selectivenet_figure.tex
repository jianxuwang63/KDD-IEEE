% Colors (muted, print-friendly)
\definecolor{c1}{RGB}{48,115,170}
\definecolor{c1bg}{RGB}{235,243,250}
\definecolor{c2}{RGB}{183,61,54}
\definecolor{c2bg}{RGB}{252,235,233}
\definecolor{c3}{RGB}{77,139,72}
\definecolor{c3bg}{RGB}{236,246,233}
\definecolor{c4}{RGB}{123,66,142}
\definecolor{c4bg}{RGB}{245,236,249}

% ============================================================
% Row 1: Standard Training
% ============================================================

% --- Case 1 ---
\node[cbox={c1}{c1bg}] (c1) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 1: Risk Control}

    \vspace{4pt}
    $\displaystyle\widehat{R}(\tau)=\frac{\sum_{i\in S_\tau}\mathbb{I}(\hat{y}_i\neq y_i)}{|S_\tau|}$

    \vspace{3pt}
    $\tau_1^*=\inf\{\tau\in[0,1]\mid\widehat{R}(\tau)\leq\alpha\}$

    \vspace{3pt}
    $(f,g)_{\tau_1^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_1^*\\[2pt]
    \text{abstain},       & \text{otherwise}
    \end{cases}$

    \vspace{6pt}
    {\footnotesize\color{c1!80!black} \textbf{Training:} Standard Cross-Entropy}\\[1pt]
    {\footnotesize\color{c1!80!black} \textbf{Calibration:} $\tau$ selected to bound risk $\leq\alpha$}
    \end{minipage}
};
\node[tag={c1}] at (c1.north) {Case 1: Risk Control};

% --- Case 2 ---
\node[cbox={c2}{c2bg}, right=1.2cm of c1] (c2) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 2: FNR Control}

    \vspace{4pt}
    $\displaystyle\widehat{\mathrm{FNR}}(\tau)=\frac{\sum_{S_\tau}\mathbb{I}(\hat{y}_i=0\wedge y_i=1)}{|\{i\in\mathcal{D}_{\mathrm{cal}}\mid y_i=1\}|}$

    \vspace{3pt}
    $\tau_2^*=\inf\{\tau\in[0,1]\mid\widehat{\mathrm{FNR}}(\tau)\leq\alpha\}$

    \vspace{3pt}
    $(f,g)_{\tau_2^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_2^*\\[2pt]
    \text{abstain},       & \text{otherwise}
    \end{cases}$

    \vspace{6pt}
    {\footnotesize\color{c2!80!black} \textbf{Training:} Standard Cross-Entropy}\\[1pt]
    {\footnotesize\color{c2!80!black} \textbf{Calibration:} $\tau$ selected to bound FNR $\leq\alpha$}
    \end{minipage}
};
\node[tag={c2}] at (c2.north) {Case 2: FNR Control};

% Horizontal arrows Row 1
\draw[bigarrow={c1}]
    (c1.east) -- node[above=2pt, font=\footnotesize, text=c1!80!black]
    {Risk-based $\tau_1^*$\hspace{6pt}vs.\hspace{6pt}FNR-based $\tau_2^*$} (c2.west);

% ============================================================
% Row 2: + Cost-Sensitive Learning
% ============================================================

% --- Case 3 ---
\node[cbox={c3}{c3bg}, below=2.0cm of c1] (c3) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 3: Risk + CSL}

    \vspace{4pt}
    {\small Same risk-based threshold $\tau_1^*$ as Case~1}

    \vspace{2pt}
    {\small\textbf{+ Cost-Sensitive Learning}}

    \vspace{4pt}
    $\displaystyle\mathcal{L}_{\mathrm{cs}}=-\frac{1}{n}\sum_{i=1}^{n}\bigl[w_1 y_i\log\hat{p}_i+w_0(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{4pt}
    {\footnotesize $w_1>1$ amplifies penalty on false negatives,}\\[1pt]
    {\footnotesize increasing sensitivity before calibration.}

    \vspace{4pt}
    {\footnotesize\color{c3!80!black} \textbf{Goal:} Control risk while suppressing FNR.}
    \end{minipage}
};
\node[tag={c3}] at (c3.north) {Case 3: Risk \& FNR (CSL)};

% --- Case 4 ---
\node[cbox={c4}{c4bg}, right=1.2cm of c3] (c4) {
    \begin{minipage}{4.6cm}\centering
    \textbf{\large Case 4: FNR + CSL}

    \vspace{4pt}
    {\small Same FNR-based threshold $\tau_2^*$ as Case~2}

    \vspace{2pt}
    {\small\textbf{+ Cost-Sensitive Learning}}

    \vspace{4pt}
    $\displaystyle\mathcal{L}_{\mathrm{cs}}=-\frac{1}{n}\sum_{i=1}^{n}\bigl[w_1 y_i\log\hat{p}_i+w_0(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{4pt}
    {\footnotesize CSL raises confidence scores on positives,}\\[1pt]
    {\footnotesize allowing a less aggressive $\tau_2^*$ at same FNR.}

    \vspace{4pt}
    {\footnotesize\color{c4!80!black} \textbf{Goal:} Recover coverage lost by Case~2.}
    \end{minipage}
};
\node[tag={c4}] at (c4.north) {Case 4: FNR \& Coverage (CSL)};
