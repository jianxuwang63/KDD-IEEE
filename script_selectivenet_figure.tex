    tag/.style n args={1}{
        rectangle, rounded corners=4pt, fill=#1, text=white,

        minimum width=6.8cm, anchor=north
    },
    harrow/.style={
        -{Latex[length=3mm,width=3mm]}, line width=1.2pt, #1

\definecolor{c1}{RGB}{48,115,170}
\definecolor{c1bg}{RGB}{235,243,250}
\definecolor{c2}{RGB}{183,61,54}
\definecolor{c2bg}{RGB}{252,235,233}
\definecolor{c3}{RGB}{77,139,72}
\definecolor{c3bg}{RGB}{236,246,233}
\definecolor{c4}{RGB}{123,66,142}
\definecolor{c4bg}{RGB}{245,236,249}

% ============================================================
% Row 1
% ============================================================

\node[cbox={c1}{c1bg}] (c1) {
    $\displaystyle\widehat{R}(\tau)=\frac{\sum_{i\in S_\tau}\mathbb{I}(\hat{y}_i\neq y_i)}{|S_\tau|}$
    \qquad
    $\tau_1^*=\inf\{\tau\in[0,1]\mid\widehat{R}(\tau)\leq\alpha\}$

    \vspace{6pt}
    $(f,g)_{\tau_1^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_1^*\\[2pt]
    \text{abstain},       & \text{otherwise}
    \end{cases}$

    \vspace{6pt}
    {\footnotesize\color{c1!80!black}
    \textbf{Training:} Standard Cross-Entropy \hspace{12pt}
    \textbf{Calibration:} $\tau$ selected to bound risk $\leq\alpha$}
};
\node[tag={c1}] at (c1.north) {Case 1: Risk Control};

\node[cbox={c2}{c2bg}, right=1.8cm of c1] (c2) {
    $\displaystyle\widehat{\mathrm{FNR}}(\tau)=\frac{\sum_{S_\tau}\mathbb{I}(\hat{y}_i=0\wedge y_i=1)}{|\{i\in\mathcal{D}_{\mathrm{cal}}\mid y_i=1\}|}$
    \qquad
    $\tau_2^*=\inf\{\tau\in[0,1]\mid\widehat{\mathrm{FNR}}(\tau)\leq\alpha\}$

    \vspace{6pt}
    $(f,g)_{\tau_2^*}(X)=
    \begin{cases}
    \mathbb{I}(f(X)>0.5), & g(X)\geq\tau_2^*\\[2pt]
    \text{abstain},       & \text{otherwise}
    \end{cases}$

    \vspace{6pt}
    {\footnotesize\color{c2!80!black}
    \textbf{Training:} Standard Cross-Entropy \hspace{12pt}
    \textbf{Calibration:} $\tau$ selected to bound FNR $\leq\alpha$}
};
\node[tag={c2}] at (c2.north) {Case 2: FNR Control};

\draw[harrow={c1}]
    (c1.east) -- (c2.west)
    node[midway, above=6pt, font=\footnotesize\bfseries, text=c1!85!black]
    {Risk-based $\tau_1^*$ \hspace{8pt}vs.\hspace{8pt} FNR-based $\tau_2^*$};

% ============================================================
% Row 2
% ============================================================

\node[cbox={c3}{c3bg}, below=1.6cm of c1] (c3) {
    {\small\textbf{Same risk-based threshold $\tau_1^*$ as Case~1 \hspace{8pt}$+$\hspace{8pt} Cost-Sensitive Learning}}

    \vspace{6pt}
    $\displaystyle\mathcal{L}_{\mathrm{cs}}=-\frac{1}{n}\sum_{i=1}^{n}
    \bigl[w_1\,y_i\log\hat{p}_i+w_0\,(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{6pt}
    {\footnotesize $w_1>1$ amplifies the penalty on false negatives, shifting the decision boundary to increase sensitivity before calibration.}

    \vspace{4pt}
    {\footnotesize\color{c3!80!black}
    \textbf{Result:} Controls risk while suppressing FNR via training-time bias toward the positive class.}
};
\node[tag={c3}] at (c3.north) {Case 3: Risk \& FNR Control (Cost-Sensitive Learning)};

\node[cbox={c4}{c4bg}, right=1.8cm of c3] (c4) {
    {\small\textbf{Same FNR-based threshold $\tau_2^*$ as Case~2 \hspace{8pt}$+$\hspace{8pt} Cost-Sensitive Learning}}

    \vspace{6pt}
    $\displaystyle\mathcal{L}_{\mathrm{cs}}=-\frac{1}{n}\sum_{i=1}^{n}
    \bigl[w_1\,y_i\log\hat{p}_i+w_0\,(1-y_i)\log(1-\hat{p}_i)\bigr]$

    \vspace{6pt}
    {\footnotesize CSL raises confidence scores on true positives, allowing a less aggressive $\tau_2^*$ at the same FNR target.}

    \vspace{4pt}
    {\footnotesize\color{c4!80!black}
    \textbf{Result:} Recovers coverage lost by Case~2 while maintaining strict FNR control.}
};
\node[tag={c4}] at (c4.north) {Case 4: FNR \& Coverage Control (Cost-Sensitive Learning)};

\draw[harrow={c3}]
    (c3.east) -- (c4.west)
    node[midway, above=6pt, font=\footnotesize\bfseries, text=c3!85!black]
    {Risk-based $\tau_1^*$ \hspace{8pt}vs.\hspace{8pt} FNR-based $\tau_2^*$};

% Vertical arrows
\draw[varrow={c3}]
    (c1.south) -- node[right=4pt, font=\footnotesize\bfseries, text=c3!80!black, pos=0.5]
    {\shortstack{+ Cost-Sensitive\\\ Learning}} (c3.north);
\draw[varrow={c4}]
    (c2.south) -- node[right=4pt, font=\footnotesize\bfseries, text=c4!80!black, pos=0.5]
    {\shortstack{+ Cost-Sensitive\\\ Learning}} (c4.north);
